create database vendaIngressos;

create table filme (
id serial primary key,
codigo varchar(15),
nome varchar(30),
genero varchar(15),
sinopse varchar(400)
);


create table sala (
id serial primary key,
nSala varchar(15),
nAssentos smallint
);

create table sessao (
id serial primary key,
qtdIngresso smallint,
horario Date,
sala smallint REFERENCES sala(id),
filme smallint REFERENCES filme(id)
);

create table vendaIngresso (
id serial primary key,
sessao smallint REFERENCES sessao(id)
);
